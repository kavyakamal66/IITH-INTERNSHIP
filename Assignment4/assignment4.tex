\section{\textbf{Question No. 2.11 - Vectors}}
\noindent Find the condition on $\vec{x}$ such that the points $\vec{x},\myvec{1\\2},\myvec{7 \\ 0}$ are collinear.
%
\section{\textbf {Solution}}
Let 
\begin{align}
 \vec{A}=\myvec{1\\2},
 \vec{B}=\myvec{7\\0}
\end{align}
The parametric equation of the line is\\
$$\mathbf{x} = \mathbf{A} + \lambda \mathbf{m}$$\\
where $\mathbf{m}$ is the direction vector and $\mathbf{A}$ is any point on the line.\\
\begin{align}
    \implies \mathbf{x}-\mathbf{A}  =  \lambda \mathbf{m} \label{eq:1}
\end{align}
Then direction vector, $\vec{m}$ of line AB is
\begin{align}
\vec{m}= \vec{B-A}=\myvec{6\\-2}
\end{align}

\begin{figure}[!h]
         \centering
         \includegraphics[width=\columnwidth]{figure4.png}
         \caption{Plot of given points}
         \label{Figure}
\end{figure}
Substituting values in \eqref{eq:1}
\begin{align}
  \vec{x}-\myvec{1\\2}=\lambda\myvec{6\\-2}\\
  \implies \vec{x}=\lambda\myvec{6\\-2}+\myvec{1\\2}
\end{align}
Now we can find $\vec{x}$ by substituting values for $\lambda$.\\
Let $\vec{C}$ be a point collinear to $\vec{A}$ and $\vec{B}$\\
For $\lambda=1/2$
\begin{align}
    \vec{C}=\myvec{4\\1}
\end{align}
Points A,B and C are plotted and is seen to be collinear.
